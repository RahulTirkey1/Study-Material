As we use PureComponents in a class. We use memo to achieve the same functionality in function components.
It was introduced in React 16.6 so it doesn't work in previous versions.
It is a higher order components.

ParentComp.js,
import React, { Component, Pure } from 'react'
import RegComp from './RegComp'
import PureComp from './PureComp'
import MemoComp from './MemoComp'

class ParentComp extends Component {
	constructor(props) {
		super(props)

		this.state = {
			name: 'Vishwas'
		}
	}

	componentDidMount() {
		setInterval(() => {
			this.setState({
				name: 'Vishwas'
			})
		}, 2000)
	}

	render() {
		console.log('*********Parent Comp render************')
		return (
			<div>
				{/* <RegComp name={this.state.name} />
				<PureComp name={this.state.name} /> */}
				<MemoComp name={this.state.name} />
			</div>
		)
	}
}

export default ParentComp;

MemoComp.js,
import React from 'react'

function MemoComp({ name }) {
  console.log('Rendering MemoComponent')
  return (
    <div>
      {name}
    </div>
  )
}

export default React.memo(MemoComp);
