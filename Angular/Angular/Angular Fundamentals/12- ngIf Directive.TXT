This is part of Structural Directives. NgIf is used to display HTML Elements based on conditions.

@Component({
  selector: 'app-test',
  template: `
    <p>
      test works!
    </p>
    <h2 *ngIf="displayName; else nameElseBlock">    //Method 1
      Codevolution
    </h2>
    <ng-template #nameElseBlock>
      <h2>
        Name is hidden
      </h2>
    </ng-template>

  
    <div *ngIf="displayName; then thenBlock; else elseBlock"></div>   //Method 2
    <ng-template #thenBlock>
      <h2>Codevolution</h2>
    </ng-template>
    <ng-template #elseBlock>
      <h2>Hidden</h2>    
    </ng-template>
  `,
  styles: []
})
export class TestComponent implements OnInit {

  public displayName = false;
  constructor() { }

  ngOnInit() {
    }
}
