A module can export classes,functions,variables etc,
Another file can import classes,functions,variables etc,

Customer.ts file,
export class Customer {

    constructor(private _firstName: string, private _lastName: string) {
    }

    public get firstName(): string {
        return this._firstName;
    }

    public set firstName(value: string) {
        this._firstName = value;
    }

    public get lastName(): string {
        return this._lastName;
    }
    public set lastName(value: string) {
        this._lastName = value;
    }
    
}

Driver.ts file,
import { Customer } from "./Customer";        //Here both the Customer.ts and Driver.ts are in the same directory.

// let's create an instance
let myCustomer = new Customer("Martin", "Dixon");

console.log(myCustomer.firstName);
console.log(myCustomer.lastName);
